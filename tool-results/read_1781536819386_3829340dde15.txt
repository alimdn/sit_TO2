   245→              {template.featured && (
   246→                <Badge className="bg-[#00D1FF]/10 text-[#00D1FF] text-xs">
   247→                  <Sparkles className="h-3 w-3 mr-1" /> Featured

   250→            </div>
   251→          )}

   253→      </div>
   254→
   255→      <div className="max-w-6xl mx-auto px-4 sm:px-6 py-8">
   256→        {template ? (
   257→          <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
   258→            {/* Left: Preview, Features, Notes, Similar Site (2 columns) */}

   261→              <div>
   262→                <h1 className="text-2xl sm:text-3xl font-bold text-[#000f22] mb-4" style={{ letterSpacing: '-0.02em' }}>
   263→                  {template.title}
   264→                </h1>
   265→                <p className="text-[#4F5B76] mb-6 leading-relaxed">{template.description}</p>

   331→                              key={i}

   333→                              className="w-full text-left px-3 py-2 rounded-lg text-xs text-[#43474d] hover:bg-[#00D1FF]/5 hover:text-[#000f22] transition-colors flex items-center gap-2"
